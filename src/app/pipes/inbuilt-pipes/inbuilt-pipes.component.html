In built pipes demo

Syntax: "3456 | currency:'USD' " Output: {{3456 | currency:'USD'}}
Syntax: "3456 | currency:'INR' " Output: {{3456 | currency:'INR'}}
Syntax: "today | date:'dd-mm-yyyy' " Output: {{ today | date:'dd-mm-yyyy'}}
Syntax: "santosh | uppercase " Output: {{ 'santosh' | uppercase}}
Syntax: "SANTOSH | uppercase " Output: {{ 'SANTOSH' | lowercase}}
Syntax: "santosh kadam | titlecase " Output: {{ 'santosh kadam' | titlecase}}
Syntax: " 0.25 | percent" Output: {{ 0.25 | percent}}
